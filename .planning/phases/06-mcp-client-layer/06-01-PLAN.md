```typescript
import { Client } from "@modelcontextprotocol/sdk/client/index.js";
import { StdioClientTransport } from "@modelcontextprotocol/sdk/client/stdio.js";

import type { McpServerConfig } from "../tools/mcp-config-loader.js";
import { log } from "../logger.js";
```

**Class: `McpClient`**

Constructor takes `McpServerConfig`. Stores:
- `readonly name: string` (from `config.name`)
- `private client: Client` (instantiated as `new Client({ name: "jarvis", version: "1.0.0" })`)
- `private transport: StdioClientTransport | StreamableHTTPClientTransport`
- `private _isAlive: boolean = false`

**Transport creation in constructor:**
- If `config.type === "stdio"`: create `StdioClientTransport` with:
  - `command: config.command!`
  - `args: config.args ?? []`
  - `env: { ...process.env, ...config.env } as Record<string, string>` (CRITICAL: merge full process.env first, then server-specific env on top. Without this, npx and other tools fail with ENOENT because PATH is missing.)
  - `stderr: "pipe"` (to capture stderr for logging)
  - After creating transport, wire `this.transport.stderr?.on("data", (chunk: Buffer) => { log("debug", \`mcp:${this.name}\`, chunk.toString().trimEnd()); })` for stderr capture.

- If `config.type === "streamable-http"`: create `StreamableHTTPClientTransport` with:
  - First arg: `new URL(config.url!)`
  - Second arg: `{ requestInit: { headers: config.headers ?? {} } }`

**Method: `async connect(): Promise<void>`**
1. Wire `this.transport.onclose` callback BEFORE calling connect (CRITICAL: SDK calls transport.start() inside connect(), so late wiring misses early events):
   ```typescript
   this.transport.onclose = () => {
     if (this._isAlive) {
       log("warn", `mcp:${this.name}`, "Server disconnected");
     }
     this._isAlive = false;
   };
   ```
2. Wire `this.transport.onerror` callback:
   ```typescript
   this.transport.onerror = (err: Error) => {
     log("error", `mcp:${this.name}`, "Transport error", { error: err.message });
   };
   ```
3. Call `await this.client.connect(this.transport);`
4. Set `this._isAlive = true;`

Note: The `onclose` callback checks `this._isAlive` before logging the warning. This prevents a spurious "Server disconnected" log during intentional shutdown (Pitfall 3 from RESEARCH.md), because `disconnect()` sets `_isAlive = false` before calling `client.close()`.

**Property: `get isAlive(): boolean`** — returns `this._isAlive`

**Method: `async listTools()`** — returns `this.client.listTools()`

**Method: `async callTool(name: string, args: Record<string, unknown>)`** — returns `this.client.callTool({ name, arguments: args })`

**Method: `async disconnect(): Promise<void>`**
1. Set `this._isAlive = false` FIRST (so onclose handler doesn't log a spurious warning)
2. Call `await this.client.close()` (SDK handles SIGTERM -> SIGKILL for stdio)

**Connection timeout is NOT in this class** — it's applied externally via `Promise.race` in index.ts (Plan 02). The McpClient class is a pure lifecycle wrapper.
